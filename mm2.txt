local function getRole(p)
if p.Character and p.Character:FindFirstChild("Knife") then
return "Murderer"
end
if p.Backpack:FindFirstChild("Knife") then
return "Murderer"
end
return "Innocent"
end

local function addESP(p)
if p == lp or drawings[p] then return end
local box = Drawing.new("Square")
box.Thickness = 2
box.Filled = false
box.Transparency = 1
local text = Drawing.new("Text")
text.Size = 16
text.Center = true
text.Outline = true
text.Font = 2
drawings[p] = {box = box, text = text}
end

for _,p in pairs(Players:GetPlayers()) do
addESP(p)
end

Players.PlayerAdded:Connect(addESP)
Players.PlayerRemoving:Connect(function(p)
if drawings[p] then
drawings[p].box:Remove()
drawings[p].text:Remove()

end
end)
